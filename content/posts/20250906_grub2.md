# grub2初体验

安装工具

    # qemu
    sudo apt install qemu-system qemu-utils
    # grub-efi-amd64 efibootmgr
    sudo apt install grub-efi-amd64 efibootmgr


虚拟磁盘

    qemu-img create -f qcow2 disk.qcow2 10G
    # nbd 驱动
    sudo modprobe nbd
    sudo qemu-nbd -c /dev/nbd0 disk.qcow2

创建分区

    sudo fdisk /dev/nbd0

    g       # gpt
    n       # new part
    1       # part num
            # start default
    +512M   # size
    t       # change part type
    1
    uefi
    a       # ? add bootable label
    1       # ?
    
    n
    2
            # start default
    +2GB

    w       # write

写入文件系统、挂载

    sudo mkfs.fat -F 32 /dev/nbd0p1
    sudo mkfs.ext4 /dev/nbd0p2

    mkdir /mnt/efi /mnt/p2 /mnt/p3
    sudo mount /dev/nbd0p1 /mnt/efi
    sudo mount /dev/nbd0p2 /mnt/p2

GRUB Install

    sudo grub-install --target x86_64-efi --efi-directory mnt/boot/efi --bootloader-id=BOOT --boot-directory=mnt/boot
    # 直接作为默认启动项（备用方式）
    mv /mnt/efi/BOOT/grubx64.efi /mnt/efi/BOOT/bootx64.efi

    # 暂时未查明原因的问题：启动一次后？或者改动了什么后，再次启动时，UEFI不会自动进入到 grub 中，而是会停留在 UEFI 固件中

写入系统

    sudo debootstrap --include=ntfs-3g,os-prober,grub-efi-amd64,grub-efi-amd64-signed,linux-image-amd64,systemd,systemd-sysv,dbus,network-manager,ifupdown,net-tools,iproute2,openssh-server stable mnt/p2/ http://mirrors.aliyun.com/debian/

    sudo mknod -m 660 /mnt/p2/dev/nbd0 b 43 0
    sudo mknod -m 660 /mnt/p2/dev/nbd0p1 b 43 1
    sudo mknod -m 660 /mnt/p2/dev/nbd0p2 b 43 2
    sudo mount --bind /dev/nbd0 /mnt/p2/dev/nbd0
    sudo mount --bind /dev/nbd0p1 /mnt/p2/dev/nbd0p1
    sudo mount --bind /dev/nbd0p2 /mnt/p2/dev/nbd0p2
    sudo mount -t proc proc /mnt/p2/proc
    sudo mount -t sysfs sysfs /mnt/p2/sys
    sudo mount -t devpts devpts mnt/p2/pts

修改配置 /etc/default/grub

    GRUB_DISABLE_OS_PROBER=false
    

切换主目录、继续处理

    sudo chroot /mnt/p2 /bin/bash

    # 或者 update-grub
    grub-mkconfig -o /boot/grub/grub.cfg

    # 需要将 /boot/grub/grub.cfg 中的 root= 的 path 改为 UUID=xxxx

启动系统

    qemu-system-x86_64 -enable-kvm -m 2G -drive if=pflash,format=raw,readonly=on,file=/usr/share/OVMF/OVMF_CODE_4M.fd \
        -drive if=pflash,format=raw,file=OVMF_VARS_4M.fd \
        -drive file=nbd.qcow2,if=ide,format=qcow2 \
        -net nic -net user -nographic


    # 排查问题
    # /boot/grub/grub.cfg
    linux 后面加上 debug earlyprintk=serial,ttyS0,115200 initcall_debug console=ttyS0,115200n8
    cat /run/initramfs/initramfs.debug
